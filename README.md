# To run server in development, run
$ npm run start:dev

# To run server test, run
$ npm run test:dev

# To run eslint test, run
$ npm run eslint

# TEMPLATES
All user interfaces design are in the template folder

# How to Run
$cd template
$npm install
$bower install

-Open the index.html in you browser

# To compile scss, run
$ gulp

# To run In Development Mode, run
$ gulp watch
